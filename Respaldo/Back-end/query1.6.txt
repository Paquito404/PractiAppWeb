CREATE TABLE Practicas (
    ID INT PRIMARY KEY IDENTITY(1,1),
    Titulo VARCHAR(100),
    Requisitos INT,
    Carrera VARCHAR(100),
    Departamento VARCHAR(100),
    Estatus VARCHAR(50),
    Integrantes INT,
    Pago INT,
    HorarioInicio TIME(0),
    HorarioFinal TIME(0),
    FechaRegistroInicio DATE,
    FechaRegistroFinal DATE,
    FechaActividadInicio DATE,
    FechaActividadFinal DATE,
    Fase INT,
    Fecha DATETIME2(0) NULL,
    Imagen VARCHAR(255),
    Autor VARCHAR(30)
);

CREATE TABLE alumnos (
	ID INT PRIMARY KEY IDENTITY(1,1),
	nombre VARCHAR(50),
	carrera VARCHAR(50),
	correo VARCHAR(50),
	semestre INT,
	password VARCHAR(50),
	ID_Practica INT,
	Fase INT
);

CREATE TABLE maestros (
	ID INT PRIMARY KEY IDENTITY(1,1),
	nombre VARCHAR(50),
	departamento VARCHAR(50),
	correo VARCHAR(50),
	password VARCHAR(50)
);

CREATE TABLE coordinadores (
	ID INT PRIMARY KEY IDENTITY(1,1),
	nombre VARCHAR(50),
	departamento VARCHAR(50),
	correo VARCHAR(50),
	password VARCHAR(50)
);

CREATE TABLE moderadores (
	ID INT PRIMARY KEY IDENTITY(1,1),
	nombre VARCHAR(50),
	departamento VARCHAR(50),
	correo VARCHAR(50),
	password VARCHAR(50)
);

CREATE TABLE Carreras (
	ID INT PRIMARY KEY IDENTITY(1,1),
	nombre VARCHAR(100),
	Departamento VARCHAR(100)
);

CREATE TABLE Departamentos (
	nombre VARCHAR(100)
);
